public class CrossLine {
	public static void main(String[] args) {
		CrossLine c = new CrossLine();
		System.out.println(c.checkCrossLine(3.14, 1, 3.14, 2));
	}
   public boolean checkCrossLine(double s1,double s2,double y1,double y2){
	   double x = (s1 - s2) / (y2 - y1);
	   double y = s1 * x + y1;
	   double yy = s2 * x + y2;
	   if(y == yy){
		   return true;
	   }else{
		   return false;
	   } 
		   
   }
}
